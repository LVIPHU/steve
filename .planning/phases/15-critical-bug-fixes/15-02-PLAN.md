### Task 2 — Convert `generateHtml()` sang streaming

**File:** `src/lib/ai-pipeline/generator.ts`

Thêm param `onChunk?: (chunk: string) => void`.

Thay `create()` → `create({ stream: true })` + `for await` loop:

```typescript
export async function generateHtml(
  userMessage: string,
  mode: "fresh" | "edit" = "fresh",
  onChunk?: (chunk: string) => void
): Promise<string> {
  const stream = await getOpenAI().chat.completions.create(
    {
      model: "gpt-4o",
      messages: [
        { role: "system", content: buildSystemPrompt(mode) },
        { role: "user", content: userMessage },
      ],
      stream: true,
    },
    { signal: AbortSignal.timeout(60000) }
  );

  let full = "";
  for await (const chunk of stream) {
    const content = chunk.choices[0]?.delta?.content ?? "";
    if (content) {
      full += content;
      onChunk?.(content);
    }
  }
  return stripMarkdownFences(full);
}
```

### Task 4 — Xử lý streaming events trong `editor-client.tsx`

**File:** `src/app/(dashboard)/dashboard/websites/[id]/edit/editor-client.tsx`

Trong `handleGenerate()` SSE event loop, thêm handler cho `status: "streaming"`:

```typescript
if (event.status === "streaming" && event.chunk) {
  // Accumulate chunks vào buffer
  streamBufferRef.current += event.chunk;
  // Update iframe srcDoc với partial HTML (throttled để không rerender quá nhiều)
  if (!streamThrottleRef.current) {
    streamThrottleRef.current = setTimeout(() => {
      setPages(prev => ({
        ...prev,
        [currentPage]: streamBufferRef.current
      }));
      streamThrottleRef.current = null;
    }, 100); // Update mỗi 100ms
  }
}
```

Thêm `streamBufferRef` và `streamThrottleRef` refs. Reset buffer khi generation bắt đầu.

Khi nhận `step: "complete"`, dùng `event.html` (final validated HTML) thay buffer.

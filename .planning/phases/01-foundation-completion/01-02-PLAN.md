```typescript
"use client";

import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { motion } from "motion/react";
import { useRouter } from "next/navigation";
import { useState } from "react";

const RESERVED_USERNAMES = [
  "dashboard", "editor", "api", "login", "register",
  "settings", "pricing", "about", "admin",
];
const USERNAME_REGEX = /^[a-z0-9][a-z0-9-]{1,28}[a-z0-9]$/;

export default function OnboardingPage() {
  const router = useRouter();
  const [username, setUsername] = useState("");
  const [error, setError] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setError(null);

    // Client-side validation (same rules as register)
    if (!USERNAME_REGEX.test(username)) {
      setError("Username chỉ được dùng chữ thường, số, dấu gạch ngang (3–30 ký tự)");
      return;
    }
    if (RESERVED_USERNAMES.includes(username.toLowerCase())) {
      setError("Username này không khả dụng");
      return;
    }

    setLoading(true);
    const res = await fetch("/api/auth/onboarding", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ username: username.toLowerCase() }),
    });
    setLoading(false);

    if (!res.ok) {
      const data = await res.json().catch(() => ({}));
      setError(data.error ?? "Không thể đặt username");
      return;
    }

    router.push("/dashboard");
    router.refresh();
  };

  return (
    <div className="flex min-h-screen items-center justify-center bg-background">
      <motion.main
        initial={{ opacity: 0, y: 10 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.3 }}
        className="w-full max-w-sm rounded-xl border border-border bg-card p-6 shadow-sm"
      >
        <h1 className="mb-2 text-xl font-semibold">Chọn username</h1>
        <p className="mb-6 text-sm text-muted-foreground">
          Username được dùng trong URL công khai của website bạn.
        </p>
        <form onSubmit={handleSubmit} className="flex flex-col gap-4">
          <div className="grid gap-2">
            <Label htmlFor="onboarding-username">Username</Label>
            <Input
              id="onboarding-username"
              type="text"
              placeholder="vd: nguyen-van-a"
              value={username}
              onChange={(e) => setUsername(e.target.value.toLowerCase())}
              required
            />
            <p className="text-xs text-muted-foreground">
              Chỉ dùng chữ thường, số, dấu gạch ngang. Tối thiểu 3 ký tự.
            </p>
          </div>
          {error && <p className="text-sm text-destructive">{error}</p>}
          <Button type="submit" disabled={loading}>
            {loading ? "Đang lưu..." : "Tiếp tục"}
          </Button>
        </form>
      </motion.main>
    </div>
  );
}
```

**File 2: `src/app/api/auth/onboarding/route.ts`**

Server-side route that creates the profiles record for authenticated users who completed Google OAuth but have no profile yet.

```typescript
import { auth } from "@/lib/auth";
import { db } from "@/db";
import { profiles } from "@/db/schema";
import { eq } from "drizzle-orm";

const RESERVED_USERNAMES = [
  "dashboard", "editor", "api", "login", "register",
  "settings", "pricing", "about", "admin",
];
const USERNAME_REGEX = /^[a-z0-9][a-z0-9-]{1,28}[a-z0-9]$/;

export async function POST(request: Request) {
  const session = await auth.api.getSession({ headers: request.headers });
  if (!session) {
    return Response.json({ error: "Unauthorized" }, { status: 401 });
  }

  const body = await request.json().catch(() => ({}));
  const username = typeof body.username === "string" ? body.username.toLowerCase() : "";

  // Server-side validation (authoritative)
  if (!USERNAME_REGEX.test(username)) {
    return Response.json({ error: "Username không hợp lệ" }, { status: 400 });
  }
  if (RESERVED_USERNAMES.includes(username)) {
    return Response.json({ error: "Username này không khả dụng" }, { status: 400 });
  }

  // Check if profile already exists (idempotency guard)
  const existing = await db
    .select()
    .from(profiles)
    .where(eq(profiles.id, session.user.id))
    .limit(1);

  if (existing.length > 0) {
    return Response.json({ error: "Profile đã tồn tại" }, { status: 409 });
  }

  try {
    await db.insert(profiles).values({
      id: session.user.id,
      username,
      plan: "free",
      createdAt: new Date(),
      updatedAt: new Date(),
    });
  } catch (err) {
    // Unique constraint violation — username taken
    const message = err instanceof Error ? err.message : "";
    if (message.includes("unique") || message.includes("duplicate")) {
      return Response.json({ error: "Username đã được sử dụng" }, { status: 409 });
    }
    return Response.json({ error: "Lỗi server" }, { status: 500 });
  }

  return Response.json({ ok: true });
}
```

```typescript
import { auth } from "@/lib/auth";
import { db } from "@/db";
import { profiles } from "@/db/schema";
import { eq } from "drizzle-orm";
import { headers } from "next/headers";
import { redirect } from "next/navigation";

export default async function DashboardLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session) redirect("/login");

  // Layer 2: Verify profile exists (username has been set)
  // If no profile → user is a Google OAuth user who hasn't completed onboarding
  const [profile] = await db
    .select({ id: profiles.id })
    .from(profiles)
    .where(eq(profiles.id, session.user.id))
    .limit(1);

  if (!profile) redirect("/onboarding");

  return <>{children}</>;
}
```

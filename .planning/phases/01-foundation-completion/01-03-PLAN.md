Create the directory and file `src/app/api/auth/mobile-token/route.ts`.

This endpoint is called by the mobile app with its own better-auth session token in the `Authorization: Bearer` header. It validates the session, creates a one-time token, stores it in the `verification` table, and returns it.

```typescript
import { auth } from "@/lib/auth";
import { db } from "@/db";
import { verification } from "@/db/schema";

export async function POST(request: Request) {
  // Validate the mobile app's session via bearer token
  // The bearer() plugin (added in Plan 01) makes auth.api.getSession()
  // read Authorization: Bearer <token> in addition to cookies
  const session = await auth.api.getSession({ headers: request.headers });

  if (!session) {
    return Response.json({ error: "Unauthorized" }, { status: 401 });
  }

  const oneTimeToken = crypto.randomUUID();
  const expiresAt = new Date(Date.now() + 5 * 60 * 1000); // 5 minutes

  // Use "mobile-token:" prefix to avoid collision with better-auth's own
  // verification entries (RESEARCH.md Pitfall 3 mitigation)
  await db.insert(verification).values({
    id: crypto.randomUUID(),
    identifier: `mobile-token:${session.user.id}`,
    value: oneTimeToken,
    expiresAt,
    createdAt: new Date(),
    updatedAt: new Date(),
  });

  return Response.json({ token: oneTimeToken });
}
```

The mobile app then constructs: `https://webapp.com/api/auth/token-login?token={oneTimeToken}` and opens it in a browser.

Create `src/app/api/auth/token-login/route.ts`.

This endpoint is opened by the mobile app in the browser. It validates the one-time token, creates a web session (with cookie), deletes the token, and redirects to `/dashboard`.

```typescript
import { auth } from "@/lib/auth";
import { db } from "@/db";
import { verification } from "@/db/schema";
import { eq, and, gt } from "drizzle-orm";
import { redirect } from "next/navigation";

export async function GET(request: Request) {
  const url = new URL(request.url);
  const token = url.searchParams.get("token");

  if (!token) {
    redirect("/login?error=invalid-token");
  }

  // Find the token — must exist and not be expired
  const [record] = await db
    .select()
    .from(verification)
    .where(
      and(
        eq(verification.value, token),
        gt(verification.expiresAt, new Date())
      )
    )
    .limit(1);

  if (!record) {
    redirect("/login?error=expired-token");
  }

  // Delete immediately — single use
  await db.delete(verification).where(eq(verification.id, record.id));

  // Extract userId from identifier (format: "mobile-token:{userId}")
  const userId = record.identifier.replace("mobile-token:", "");
```

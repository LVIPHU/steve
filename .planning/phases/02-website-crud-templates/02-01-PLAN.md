    ```typescript
      { id: "blog",      name: "Blog",      icon: "\u{1F4DD}" },
      { id: "portfolio", name: "Portfolio",  icon: "\u{1F4BC}" },
      { id: "fitness",   name: "Fitness",    icon: "\u{1F4AA}" },
      { id: "cooking",   name: "Cooking",    icon: "\u{1F373}" },
      { id: "learning",  name: "Learning",   icon: "\u{1F393}" },
    ] as const;

    export type TemplateId = typeof TEMPLATES[number]["id"];

    const KEYWORD_MAP: Record<string, TemplateId> = {
      blog: "blog", bai: "blog", viet: "blog", post: "blog", article: "blog",
      portfolio: "portfolio", project: "portfolio", work: "portfolio",
      fitness: "fitness", gym: "fitness", workout: "fitness", exercise: "fitness", suc: "fitness",
      cooking: "cooking", nau: "cooking", mon: "cooking", recipe: "cooking",
      learning: "learning", hoc: "learning", khoa: "learning", course: "learning", study: "learning",
    };

    export function suggestTemplate(input: string): TemplateId | null {
      if (!input) return null;
      const lower = input.toLowerCase();
      for (const [keyword, templateId] of Object.entries(KEYWORD_MAP)) {
        if (lower.includes(keyword)) return templateId;
      }
      return null;
    }
    ```

    5. Create src/lib/slug.ts:
    ```typescript
    export function generateSlug(name: string): string {
      const slug = name
        .toLowerCase()
        .trim()
        .replace(/[^\w\s-]/g, "")
        .replace(/[\s_]+/g, "-")
        .replace(/^-+|-+$/g, "")
        .substring(0, 60);
      return slug || "website";
    }
    ```

    Create src/app/api/websites/[id]/route.ts with two handlers:

    **PATCH handler** — accepts JSON body with either `{ name: string }` or `{ status: "draft" | "published" | "archived" }`:
    1. Get session via `auth.api.getSession({ headers: request.headers })`
    2. Return 401 if no session
    3. Parse `params` — in Next.js 15+, params is a Promise: `const { id } = await params;`
    4. Query website with ownership check: `db.select().from(websites).where(and(eq(websites.id, id), eq(websites.userId, session.user.id))).limit(1)`
    5. Return 404 if not found
    6. If body has `name`: validate non-empty after trim, update `{ name: body.name.trim(), updatedAt: new Date() }`
    7. If body has `status`: validate value is one of "draft", "published", "archived", update `{ status: body.status, updatedAt: new Date() }`
    8. Return `Response.json({ ok: true })`

    **DELETE handler:**
    1. Same session + ownership check as PATCH
    2. `db.delete(websites).where(and(eq(websites.id, id), eq(websites.userId, session.user.id)))`
    3. Return `Response.json({ ok: true })`

    Import pattern:
    ```typescript
    import { auth } from "@/lib/auth";
    import { db } from "@/db";
    import { websites } from "@/db/schema";
    import { and, eq } from "drizzle-orm";
    ```

    Type for params in Next.js 15+:
    ```typescript
    type RouteParams = { params: Promise<{ id: string }> };
    export async function PATCH(request: Request, { params }: RouteParams) { ... }
    export async function DELETE(request: Request, { params }: RouteParams) { ... }
    ```

    IMPORTANT: Do NOT validate the `id` format as UUID since Drizzle handles that. Do NOT wrap redirect() in try/catch (no redirects in API routes anyway).

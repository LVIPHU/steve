```typescript
import { auth } from "@/lib/auth";
import { db } from "@/db";
import { websites } from "@/db/schema";
import { eq, desc } from "drizzle-orm";

import { redirect } from "next/navigation";
import Link from "next/link";
import DashboardNav from "../dashboard-nav";
import WebsiteCard from "@/components/website-card";
import { Button } from "@/components/ui/button";

```

     - Right: `<Link href="/dashboard/websites/new"><Button>Tao website moi</Button></Link>`

4. **Empty state** (when `userWebsites.length === 0`):
   ```tsx
   <div className="flex flex-col items-center justify-center py-16 text-center">

     <h2 className="text-xl font-semibold mb-2">Ban chua co website nao</h2>
   ```

   ```tsx
   <div className="grid grid-cols-2 gap-4 md:grid-cols-3 md:gap-6">
     {userWebsites.map((site, index) => (
       <WebsiteCard key={site.id} website={site} index={index} />
     ))}
   </div>
   ```
   Pass `index` prop to WebsiteCard for stagger animation timing.

NOTE: The page itself is a Server Component. The `WebsiteCard` is a Client Component imported here. Server Components can import and render Client Components by passing serializable props. The `website` object from Drizzle select is serializable (plain object with primitives + dates).

IMPORTANT: `motion` cannot be used in Server Components. The stagger animation is handled inside the WebsiteCard Client Component using the `index` prop for delay calculation.

**Module-level components (NOT inside WebsiteCard):**

1. **StatusBadge** — same as in detail page:
```typescript
function StatusBadge({ status }: { status: string }) {
  return (
    <span
      aria-label={`Trang thai: ${status}`}
      className={cn(
        "rounded-full px-2 py-0.5 text-xs font-medium",
        status === "published" && "bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400",
        status === "draft" && "bg-gray-100 text-gray-600 dark:bg-gray-800 dark:text-gray-400",
        status === "archived" && "bg-orange-100 text-orange-700 dark:bg-orange-900/30 dark:text-orange-400",
      )}
    >
      {status === "published" ? "Published" : status === "draft" ? "Draft" : "Archived"}
    </span>
  );
}
```

**WebsiteCard component structure:**

1. **Wrapper:** `motion.div` with stagger entrance animation:
   - `initial={{ opacity: 0, y: 8 }}`
   - `animate={{ opacity: 1, y: 0 }}`
   - `transition={{ duration: 0.2, delay: index * 0.04 }}`

2. **Card:** Uses `Card` component with `group` class for hover detection:
   ```tsx
   <Card className="relative group hover:border-border/80 hover:shadow-md transition-shadow cursor-pointer">
   ```

3. **Card click navigation:** The card body is wrapped in `<Link href={`/dashboard/websites/${website.id}`}>` but the menu button area must use `e.stopPropagation()` and `e.preventDefault()` to prevent navigation when interacting with dropdown.

4. **Three-dot menu button (hover reveal):**
   ```tsx
   <Button
     variant="ghost"
     size="icon"
     className="absolute top-2 right-2 opacity-0 group-hover:opacity-100 transition-opacity z-10"
     onClick={(e) => { e.preventDefault(); e.stopPropagation(); setMenuOpen(!menuOpen); }}
     aria-label="Mo menu"
     aria-expanded={menuOpen}
   >
     <MoreVertical className="size-4" />
   </Button>
   ```

5. **Dropdown menu** (custom, no library):
   When `menuOpen` is true, render an absolutely positioned div:
   ```tsx
   <motion.div
     initial={{ opacity: 0, scale: 0.95 }}
     animate={{ opacity: 1, scale: 1 }}
     transition={{ duration: 0.1 }}
     className="absolute top-10 right-2 z-50 min-w-[140px] bg-popover border border-border rounded-lg shadow-md py-1"
     role="menu"
   >
   ```
   Menu items (each `role="menuitem"`, `px-3 py-1.5 text-sm cursor-pointer hover:bg-muted w-full text-left`):
   - "Doi ten" — onClick: `setRenaming(true); setMenuOpen(false);`
   - "Doi trang thai" — onMouseEnter: `setStatusMenuOpen(true)`, shows sub-menu
     - Sub-menu: three buttons for "Draft", "Published", "Archived" (each `Button variant="ghost" size="sm"`)
     - onClick for each: call `handleStatusChange(newStatus)`
   - "Xoa" — onClick: `setConfirmDelete(true); setMenuOpen(false);` (text-destructive color)

6. **Click outside to close menu:** Use a `useRef` on the menu container and `useEffect` with `mousedown` event listener on `document`. When click is outside ref, close menu. Clean up listener on unmount.

7. **Card content (normal mode):**
   ```tsx
   <CardHeader className="pb-2">
     <CardTitle className="text-sm">{website.name}</CardTitle>
   </CardHeader>
   <CardContent className="pt-0">
     <StatusBadge status={website.status} />
   </CardContent>
   ```

8. **Inline rename mode** (when `renaming` is true):
   Replace CardTitle with Input:
   ```tsx
   <Input
     value={renameValue}
     onChange={(e) => setRenameValue(e.target.value)}
     onKeyDown={handleRenameKeyDown}
     onBlur={() => { setRenaming(false); setRenameValue(website.name); }}
     autoFocus
     aria-label="Doi ten website"
     className="text-sm h-7"
     onClick={(e) => { e.preventDefault(); e.stopPropagation(); }}
   />
   ```
   - `handleRenameKeyDown`: Enter key → call `handleRename()`, Escape → cancel (restore original name, exit rename mode)
   - `handleRename`: `fetch(`/api/websites/${website.id}`, { method: "PATCH", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ name: renameValue.trim() }) })` → if ok, `router.refresh()`, set `renaming(false)`. Show toast on success (simple approach: no toast library in Phase 2, just call `router.refresh()` to reflect new name).

9. **Delete confirm mode** (when `confirmDelete` is true):
   Show inline panel below card content:
   ```tsx
   <div className="px-6 pb-4">
     <div className="bg-destructive/5 border border-destructive/20 rounded-lg p-3">
       <p className="text-sm font-medium mb-1">Xoa website nay?</p>
       <p className="text-xs text-muted-foreground mb-3">Hanh dong nay khong the hoan tac.</p>
       <div className="flex gap-2 justify-end">
         <Button variant="outline" size="sm" onClick={(e) => { e.preventDefault(); e.stopPropagation(); setConfirmDelete(false); }}>Huy</Button>
         <Button variant="destructive" size="sm" disabled={loading} onClick={(e) => { e.preventDefault(); e.stopPropagation(); handleDelete(); }}>Xoa</Button>
       </div>
     </div>
   </div>
   ```
   - `handleDelete`: set loading, `fetch(`/api/websites/${website.id}`, { method: "DELETE" })` → if ok, `router.refresh()`.

10. **handleStatusChange(newStatus: string):**
    ```typescript
    async function handleStatusChange(newStatus: string) {
      setMenuOpen(false);
      setStatusMenuOpen(false);
      await fetch(`/api/websites/${website.id}`, {
        method: "PATCH",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ status: newStatus }),
      });
      router.refresh();
    }
    ```

IMPORTANT per rerender-no-inline-components: StatusBadge is defined at module level. The dropdown menu items can be inline JSX within the component since they are simple elements, not components.

IMPORTANT: All click handlers on interactive elements inside the card Link must call `e.preventDefault(); e.stopPropagation();` to prevent navigation.

```typescript
import OpenAI from "openai";
import { buildSystemPrompt, stripMarkdownFences } from "@/lib/html-prompts";

let _openai: OpenAI | null = null;
function getOpenAI(): OpenAI {
  if (!_openai) _openai = new OpenAI();
  return _openai;
}

export async function generateHtml(
  userMessage: string,
  currentHtml?: string
): Promise<string> {
  const messages: { role: "system" | "user"; content: string }[] = [
    { role: "system", content: buildSystemPrompt() },
    { role: "user", content: currentHtml
      ? `${userMessage}\n\n## Current HTML to Edit\n${currentHtml}`
      : userMessage
    },
  ];

  const completion = await getOpenAI().chat.completions.create(
    { model: "gpt-4o", messages },
    { signal: AbortSignal.timeout(60000) }
  );

  const raw = completion.choices[0].message.content ?? "";
  return stripMarkdownFences(raw);
}
```

Key decisions:
- For edit mode: `currentHtml` is appended to the user message as a `## Current HTML to Edit` section. This resolves the Open Question from 11-RESEARCH.md — the orchestrator calls `buildEditUserMessage(prompt)` which returns preservation instruction + request, then generator appends currentHtml to that user message. The LLM sees: system prompt (lean) + user message (preservation + request + current HTML).
- Lazy `getOpenAI()` pattern replaces module-scope `const openai = new OpenAI()` — fixes test import failures.
- `buildFreshSystemPrompt` and `buildEditSystemPrompt` are no longer imported.

**1. Rewrite `src/lib/ai-pipeline/index.ts` completely:**

Replace with the 7-step fresh / 4-step edit orchestrator:

```typescript
import { analyzePrompt } from "./analyzer";
import { generateHtml } from "./generator";
import { validateAndFix } from "./validator";
import { runDesignAgent } from "./design-agent";
import { reviewHtml, getReviewThreshold, shouldTriggerRefine, appendCalibrationLog } from "./reviewer";
import { buildUserMessage, buildEditUserMessage, refineHtml } from "./context-builder";
import { selectComponents } from "@/lib/component-library";
import type { PipelineEvent } from "./types";

export type { PipelineEvent } from "./types";

const ENABLE_REFINE = process.env.ENABLE_REFINE === "true";

export async function runGenerationPipeline(
  prompt: string,
  currentHtml: string | undefined,
  onEvent: (event: PipelineEvent) => void
): Promise<string> {
  const isEdit = !!currentHtml;

  // Step 1: Analyze (always)
  onEvent({ step: "analyze", status: "start" });
  const analysis = await analyzePrompt(prompt);
  onEvent({
    step: "analyze",
    status: "done",
    detail: `Loai: ${analysis.type} · Sections: ${analysis.sections.slice(0, 4).join(", ") || "auto"}`,
  });

  // Step 2: Components (always)
  onEvent({ step: "components", status: "start" });
  const snippets = selectComponents(analysis);
  onEvent({
    step: "components",
    status: "done",
    detail: `${snippets.length} components`,
  });

  // Step 3: Design (fresh only)
  let design;
  if (!isEdit) {
    onEvent({ step: "design", status: "start" });
    design = await runDesignAgent(prompt, analysis);
    onEvent({ step: "design", status: "done", detail: design.preset });
  }

  // Step 4: Generate
  onEvent({ step: "generate", status: "start" });
  const userMessage = isEdit
    ? buildEditUserMessage(prompt)
    : buildUserMessage(prompt, analysis, design!, snippets);
  let html = await generateHtml(userMessage, currentHtml);
  onEvent({ step: "generate", status: "done" });

  // Step 5-6: Review + Refine (fresh + ENABLE_REFINE only)
  if (!isEdit && ENABLE_REFINE) {
    onEvent({ step: "review", status: "start" });
    const reviewResult = await reviewHtml(html, prompt);
    const threshold = getReviewThreshold();
    const triggerRefine = shouldTriggerRefine(reviewResult, threshold);
    onEvent({
      step: "review",
      status: "done",
      detail: `Score: ${reviewResult.score}/100`,
    });

    if (triggerRefine) {
      onEvent({ step: "refine", status: "start" });
      html = await refineHtml(html, reviewResult);
      onEvent({ step: "refine", status: "done" });
    }

    // Always log calibration when review runs
    appendCalibrationLog({
      prompt,
      reviewResult,
      triggered_refine: triggerRefine,
    });
  }

  // Step 7: Validate (always)
  onEvent({ step: "validate", status: "start" });
  const { html: finalHtml, fixes, warnings } = validateAndFix(html);
  const fixDetail =
    fixes.length > 0
      ? `${fixes.length} fix(es) applied`
      : warnings.length > 0
      ? `${warnings.length} warning(s)`
      : "Khong phat hien loi";
  onEvent({ step: "validate", status: "done", detail: fixDetail, fix_count: fixes.length });

  if (warnings.length > 0) {
    console.warn("[ai-pipeline] validator warnings:", warnings);
  }

  return finalHtml;
}
```

Key points:
- `researchContext` import removed — no more researcher.ts
- `ENABLE_REFINE` read from env at module level (default: false = Hobby-safe)
- Edit mode: 4 steps (analyze, components, generate, validate) — design/review/refine skipped
- Fresh mode with ENABLE_REFINE=true: 7 steps (analyze, components, design, generate, review, refine conditional, validate)
- Fresh mode with ENABLE_REFINE=false: 5 steps (analyze, components, design, generate, validate)
- Calibration log appended after every review (regardless of refine trigger)

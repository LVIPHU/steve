    ```typescript
    export function buildGoogleFontsImport(fonts: { heading: string; body: string }): string {
      const families = [fonts.heading, fonts.body]
        .filter((f, i, arr) => arr.indexOf(f) === i) // deduplicate if same font
        .map((name) => `family=${name.replace(/ /g, "+")}:wght@400;600;700`)
        .join("&");
      return `@import url('https://fonts.googleapis.com/css2?${families}&display=swap');`;
    }
    ```

    ```typescript
    export async function generateHtml(
      prompt: string,
      analysis: AnalysisResult,
      research: ResearchResult,        // kept for backward compat — Phase 11 removes
      currentHtml?: string,
      design?: DesignResult,           // NEW — optional for backward compat
      snippets?: ComponentSnippet[]    // NEW — optional for backward compat
    ): Promise<string> {
      const systemPrompt = buildSystemPrompt();

      let userMessage: string;
      if (currentHtml) {
        userMessage = buildEditUserMessage(prompt);
        userMessage = `Current HTML:\n${currentHtml}\n\n${userMessage}`;
      } else if (design && snippets) {
        userMessage = buildUserMessage(prompt, analysis, design, snippets);
      } else {
        userMessage = prompt;
      }

      const completion = await openai.chat.completions.create(
        {
          model: "gpt-4o",
          messages: [
            { role: "system", content: systemPrompt },
            { role: "user", content: userMessage },
          ],
        },
        { signal: AbortSignal.timeout(60000) }
      );

      const raw = completion.choices[0].message.content ?? "";
      return stripMarkdownFences(raw);
    }
    ```

**handleGenerate function:**
```typescript
async function handleGenerate(prompt: string) {
  setIsGenerating(true);
  // Add user message
  setMessages(prev => [...prev, { role: "user", content: prompt, timestamp: new Date() }]);

  try {
    const res = await fetch(`/api/ai/generate-html`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        websiteId: props.websiteId,
        prompt,
        currentHtml: htmlContent || undefined,
      }),
    });

    if (!res.ok) throw new Error("Generation failed");
    const data = await res.json();

    setHtmlContent(data.html);
    setCodeValue(data.html);
    setMessages(prev => [...prev, { role: "assistant", content: "Đã cập nhật!", timestamp: new Date() }]);
    setHasUnsaved(false); // API already saved it
    scheduleAutoSave(data.html); // Redundant save for safety
  } catch {
    setMessages(prev => [...prev, { role: "error", content: "Không thể tạo nội dung. Vui lòng thử lại.", timestamp: new Date() }]);
  } finally {
    setIsGenerating(false);
  }
}
```

**handleSave function:**
```typescript
async function handleSave(html?: string) {
  setIsSaving(true);
  try {
    const res = await fetch(`/api/websites/${props.websiteId}`, {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ html_content: html ?? htmlContent }),
    });
    if (!res.ok) throw new Error();
    setHasUnsaved(false);
    toast("Đã lưu!");
  } catch {
    toast.error("Lưu thất bại, thử lại.");
  } finally {
    setIsSaving(false);
  }
}
```

**handlePublish function:**
```typescript
async function handlePublish() {
  const res = await fetch(`/api/websites/${props.websiteId}`, {
    method: "PATCH",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ status: "published" }),
  });
  if (res.ok) {
    setStatus("published");
    toast("Đã xuất bản!");
  }
}
```

**Auto-save (500ms debounce after generation):**
```typescript
function scheduleAutoSave(html: string) {
  if (saveTimeoutRef.current) clearTimeout(saveTimeoutRef.current);
  saveTimeoutRef.current = setTimeout(() => { void handleSave(html); }, 500);
}
```
